# Лема
$$
\begin{flalign}
    & \forall x, u, v \in V^n, u \not= v:&&\\
    & 1)\ x^u \land x^v = 0\\
    & 2)\ x^u \lor x^v = x^u \oplus x^v\\
    & \\
    &\text{Доведення:}\\
    &\text{Нехай } x = x_1x_2x_3...x_n\\
    &\phantom{\text{Нехай }} u = u_1u_2u_3...u_n\\
    &\phantom{\text{Нехай }} v = v_1v_2v_3...v_n\\
    & \text{Тоді:}\\
    & 1)\ x^u \land x^v = x_1^{u_1}x_1^{v_1} x_2^{u_2}x_2^{v_2}...x_n^{u_n}x_n^{v_n}\\
    & 2)\ x^u \lor x^v = x_1^{u_1}x_2^{u_2}...x_n^{u_n} \lor x_1^{v_1}x_2^{v_2}...x_n^{v_n},\\
    & x^u \oplus x^v = x_1^{u_1}x_2^{u_2}...x_n^{u_n} \oplus x_1^{v_1}x_2^{v_2}...x_n^{v_n}\\
    &\\
    & 1)\ u \not= v \Rightarrow \exists i: u_i \not= v_i \Rightarrow
    \exists x_i^0 x_i^1 = \overline{x_i} x_i = 0 \land 1 = 0 \text{ серед кон'юнкцій. Отже вся кон'юнкція рівна 0} \Rightarrow \\
    & \Rightarrow x^u \land x^v = 0\\
    & 2)\\
    & \text{ -Якщо } x = v(\text{або } x = u, \text{ аналогічно}):\\
    & x^u \lor x^v = x_1^{u_1} x_2^{u_2}...x_n^{u_n} \lor x^v \text{ або, якщо } x = v,\ 
    x^u \lor x^v = x^u \lor x_1^{v_1} x_2^{v_2}...x_n^{v_n}\\
    & \text{Одне з них складається з } x_i^{x_i}\text{. Якщо } x_i = 0, x_i^{x_i} = 0^0 = 1\text{, якщо } x_i = 1, x_i^{x_i} = 1^1 = 1\\
    & \text{Отже, одна з } \{x^v, x^u\} \text{ буде рівна 1. При чому з 1)} \Rightarrow x^u \land x^v = 0 \Rightarrow
    \text{Інша з них рівна 0. Тобто}\\
    & \bcase{x^v = 1, x^u = 0 \\
    x^u = 1, x^v = 0} \Rightarrow
    \begin{cases}
        x^u \lor x^v = 1\\
        x^u \oplus x^v = 1
    \end{cases} - \heartsuit\\
    & \text{ -Якщо } x \not= v \text{ і } x \not= u:\\
    & \exists i \exists j: x_i \not= u_i, x_j \not= v_j \Rightarrow x_i^{u_i} = 0, x_j^{v_j} = 0 \Rightarrow \text{ серед кожної з кон'юнкцій } \l\{ x^u, x^v \r\}
    \text{ існує 0, а отже кожна кон'юнкція рівна 0} \Rightarrow\\
    & \Rightarrow \begin{cases}
        x^u \lor x^v = 0\\
        x^u \oplus x^v = 0
    \end{cases} - \spadesuit\\
    & \heartsuit \land \spadesuit \Rightarrow x^u \lor x^v = x^u \oplus x^v\\
\end{flalign}
$$
