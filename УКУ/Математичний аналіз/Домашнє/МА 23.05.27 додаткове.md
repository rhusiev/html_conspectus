# 1
$$
\begin{flalign}
    & f_{X,Y} (x, y) = c * \iint_{\substack{4 \le x < 5 \\ 0 \le y < \infty}} x \cdot e^{- 2y} dxdy = 
    = c * \int_{0}^{\infty} dy \int_{4}^{5} x \cdot e^{- 2y} dx = &&\\
    & = c * \int_{0}^{\infty} dy \l( e^{- 2y} \frac{x^2}{2} \r) \|_{4}^{5} =
    c * \int_{0}^{\infty} e^{- 2y} \frac{9}{2} dy = c * \frac{9}{2} \frac{-1}{2} e^{- 2y} \|_{0}^{\infty} = &&\\
    & = \frac{- 9 \cdot c}{4} \cdot (-1) = \frac{9 \cdot c}{4} = 1 \\
    & c = \frac{4}{9} = 0.4444 \\
\end{flalign}
$$
